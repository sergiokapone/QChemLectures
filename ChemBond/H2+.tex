\section{–¢–µ–æ—Ä–µ—Ç–∏—á–µ—Å–∫–∏–µ –æ—Å–Ω–æ–≤—ã}

\subsection{–ü–æ—Å—Ç–∞–Ω–æ–≤–∫–∞ –∑–∞–¥–∞—á–∏}

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø—Ä–æ—Å—Ç–µ–π—à—É—é –º–æ–ª–µ–∫—É–ª—è—Ä–Ω—É—é —Å–∏—Å—Ç–µ–º—É -- –∏–æ–Ω –≤–æ–¥–æ—Ä–æ–¥–∞ H$_2^+$, —Å–æ—Å—Ç–æ—è—â–∏–π –∏–∑ –¥–≤—É—Ö –ø—Ä–æ—Ç–æ–Ω–æ–≤ –∏ –æ–¥–Ω–æ–≥–æ —ç–ª–µ–∫—Ç—Ä–æ–Ω–∞.

\begin{theorembox}{–û—Å–Ω–æ–≤–Ω–∞—è –∏–¥–µ—è –ú–û-–õ–ö–ê–û}
–ú–æ–ª–µ–∫—É–ª—è—Ä–Ω–∞—è –æ—Ä–±–∏—Ç–∞–ª—å $\psi$ –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç—Å—è –≤ –≤–∏–¥–µ –ª–∏–Ω–µ–π–Ω–æ–π –∫–æ–º–±–∏–Ω–∞—Ü–∏–∏ –∞—Ç–æ–º–Ω—ã—Ö –æ—Ä–±–∏—Ç–∞–ª–µ–π:
\begin{equation}
\psi = c_1\phi_1 + c_2\phi_2 + \ldots + c_N\phi_N = \sum_{i=1}^{N} c_i\phi_i
\end{equation}
–≥–¥–µ $\phi_i$ -- –∞—Ç–æ–º–Ω—ã–µ –æ—Ä–±–∏—Ç–∞–ª–∏, $c_i$ -- –∫–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç—ã —Ä–∞–∑–ª–æ–∂–µ–Ω–∏—è.
\end{theorembox}

–î–ª—è –∏–æ–Ω–∞ H$_2^+$ –≤ –º–∏–Ω–∏–º–∞–ª—å–Ω–æ–º –±–∞–∑–∏—Å–µ:
\begin{equation}
\psi = c_1\phi_1 + c_2\phi_2
\end{equation}
–≥–¥–µ $\phi_1$ –∏ $\phi_2$ -- 1s –æ—Ä–±–∏—Ç–∞–ª–∏ –≤–æ–¥–æ—Ä–æ–¥–∞ –Ω–∞ —è–¥—Ä–∞—Ö A –∏ B —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ.

\subsection{–í–∞—Ä–∏–∞—Ü–∏–æ–Ω–Ω—ã–π –ø—Ä–∏–Ω—Ü–∏–ø}

\begin{theorembox}{–í–∞—Ä–∏–∞—Ü–∏–æ–Ω–Ω—ã–π –ø—Ä–∏–Ω—Ü–∏–ø –†—ç–ª–µ—è-–†–∏—Ç—Ü–∞}
–î–ª—è –ª—é–±–æ–π –Ω–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–Ω–æ–π –ø—Ä–æ–±–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ $\psi_{\text{–ø—Ä–æ–±–Ω}}$ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è:
\begin{equation}
E_{\text{–ø—Ä–æ–±–Ω}} = \frac{\langle\psi_{\text{–ø—Ä–æ–±–Ω}}|\hat{H}|\psi_{\text{–ø—Ä–æ–±–Ω}}\rangle}{\langle\psi_{\text{–ø—Ä–æ–±–Ω}}|\psi_{\text{–ø—Ä–æ–±–Ω}}\rangle} \geq E_{\text{—Ç–æ—á–Ω}}
\end{equation}
–≥–¥–µ $E_{\text{—Ç–æ—á–Ω}}$ -- —Ç–æ—á–Ω–∞—è —ç–Ω–µ—Ä–≥–∏—è –æ—Å–Ω–æ–≤–Ω–æ–≥–æ —Å–æ—Å—Ç–æ—è–Ω–∏—è.
\end{theorembox}

–≠—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç, —á—Ç–æ –≤–∞—Ä—å–∏—Ä—É—è –ø–∞—Ä–∞–º–µ—Ç—Ä—ã –ø—Ä–æ–±–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ –∏ –º–∏–Ω–∏–º–∏–∑–∏—Ä—É—è —ç–Ω–µ—Ä–≥–∏—é, –º—ã –ø–æ–ª—É—á–∞–µ–º –Ω–∞–∏–ª—É—á—à–µ–µ –ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏–µ –≤ —Ä–∞–º–∫–∞—Ö –≤—ã–±—Ä–∞–Ω–Ω–æ–≥–æ —Ñ—É–Ω–∫—Ü–∏–æ–Ω–∞–ª—å–Ω–æ–≥–æ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–∞.

\section{–í—ã–≤–æ–¥ —Å–µ–∫—É–ª—è—Ä–Ω–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è}

\subsection{–§—É–Ω–∫—Ü–∏–æ–Ω–∞–ª —ç–Ω–µ—Ä–≥–∏–∏}

–î–ª—è –õ–ö–ê–û —Ñ—É–Ω–∫—Ü–∏–∏ $\psi = c_1\phi_1 + c_2\phi_2$ —ç–Ω–µ—Ä–≥–∏—è –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è –∫–∞–∫:
\begin{equation}
E = \frac{\langle\psi|\hat{H}|\psi\rangle}{\langle\psi|\psi\rangle} = \frac{c_1^2H_{11} + c_2^2H_{22} + 2c_1c_2H_{12}}{c_1^2S_{11} + c_2^2S_{22} + 2c_1c_2S_{12}}
\end{equation}

–≥–¥–µ –≤–≤–µ–¥–µ–Ω—ã –æ–±–æ–∑–Ω–∞—á–µ–Ω–∏—è:
\begin{align}
H_{ij} &= \langle\phi_i|\hat{H}|\phi_j\rangle \quad \text{(–º–∞—Ç—Ä–∏—á–Ω—ã–µ —ç–ª–µ–º–µ–Ω—Ç—ã –≥–∞–º–∏–ª—å—Ç–æ–Ω–∏–∞–Ω–∞)} \\
S_{ij} &= \langle\phi_i|\phi_j\rangle \quad \text{(–∏–Ω—Ç–µ–≥—Ä–∞–ª—ã –ø–µ—Ä–µ–∫—Ä—ã–≤–∞–Ω–∏—è)}
\end{align}

\subsection{–£—Å–ª–æ–≤–∏—è –º–∏–Ω–∏–º—É–º–∞}

–ü—Ä–∏–º–µ–Ω—è—è —É—Å–ª–æ–≤–∏—è —ç–∫—Å—Ç—Ä–µ–º—É–º–∞:
\begin{equation}
\frac{\partial E}{\partial c_1} = 0, \quad \frac{\partial E}{\partial c_2} = 0
\end{equation}

–ü–æ—Å–ª–µ –∞–ª–≥–µ–±—Ä–∞–∏—á–µ—Å–∫–∏—Ö –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–π –ø–æ–ª—É—á–∞–µ–º —Å–∏—Å—Ç–µ–º—É –ª–∏–Ω–µ–π–Ω—ã—Ö —É—Ä–∞–≤–Ω–µ–Ω–∏–π:
\begin{align}
c_1(H_{11} - ES_{11}) + c_2(H_{12} - ES_{12}) &= 0 \\
c_1(H_{21} - ES_{21}) + c_2(H_{22} - ES_{22}) &= 0
\end{align}

\begin{theorembox}{–°–µ–∫—É–ª—è—Ä–Ω–æ–µ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ}
–°–∏—Å—Ç–µ–º–∞ –∏–º–µ–µ—Ç –Ω–µ—Ç—Ä–∏–≤–∏–∞–ª—å–Ω–æ–µ —Ä–µ—à–µ–Ω–∏–µ —Ç–æ–ª—å–∫–æ –ø—Ä–∏ —É—Å–ª–æ–≤–∏–∏:
\begin{equation}
\begin{vmatrix}
H_{11} - ES_{11} & H_{12} - ES_{12} \\
H_{21} - ES_{21} & H_{22} - ES_{22}
\end{vmatrix} = 0
\end{equation}
\end{theorembox}

\section{–†–µ—à–µ–Ω–∏–µ –¥–ª—è H$_2^+$}

\subsection{–ú–∞—Ç—Ä–∏—á–Ω—ã–µ —ç–ª–µ–º–µ–Ω—Ç—ã}

–î–ª—è —Å–∏–º–º–µ—Ç—Ä–∏—á–Ω–æ–π —Å–∏—Å—Ç–µ–º—ã H$_2^+$:
\begin{align}
H_{11} = H_{22} &= \alpha \quad \text{(–∫—É–ª–æ–Ω–æ–≤—Å–∫–∏–µ –∏–Ω—Ç–µ–≥—Ä–∞–ª—ã)} \\
H_{12} = H_{21} &= \beta \quad \text{(—Ä–µ–∑–æ–Ω–∞–Ω—Å–Ω—ã–µ –∏–Ω—Ç–µ–≥—Ä–∞–ª—ã)} \\
S_{11} = S_{22} &= 1 \quad \text{(–Ω–æ—Ä–º–∏—Ä–æ–≤–∫–∞)} \\
S_{12} = S_{21} &= S \quad \text{(–∏–Ω—Ç–µ–≥—Ä–∞–ª –ø–µ—Ä–µ–∫—Ä—ã–≤–∞–Ω–∏—è)}
\end{align}

\subsection{–°–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è —ç–Ω–µ—Ä–≥–∏–∏}

–°–µ–∫—É–ª—è—Ä–Ω–æ–µ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –≤–∏–¥:
\begin{equation}
\begin{vmatrix}
\alpha - E & \beta - ES \\
\beta - ES & \alpha - E
\end{vmatrix} = 0
\end{equation}

–†–∞—Å–∫—Ä—ã–≤–∞—è –æ–ø—Ä–µ–¥–µ–ª–∏—Ç–µ–ª—å:
\begin{equation}
(\alpha - E)^2 - (\beta - ES)^2 = 0
\end{equation}

\begin{examplebox}{–†–µ—à–µ–Ω–∏–µ –∫–≤–∞–¥—Ä–∞—Ç–Ω–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è}
\begin{align}
\alpha - E &= \pm(\beta - ES) \\
E_{1,2} &= \frac{\alpha \pm \beta}{1 \pm S}
\end{align}

\textbf{–î–≤–∞ —Ä–µ—à–µ–Ω–∏—è:}
\begin{itemize}
\item \textbf{–°–≤—è–∑—ã–≤–∞—é—â–∞—è –ú–û:} $E_1 = \frac{\alpha + \beta}{1 + S}$ (–Ω–∏–∂–Ω—è—è —ç–Ω–µ—Ä–≥–∏—è)
\item \textbf{–ê–Ω—Ç–∏—Å–≤—è–∑—ã–≤–∞—é—â–∞—è –ú–û:} $E_2 = \frac{\alpha - \beta}{1 - S}$ (–≤–µ—Ä—Ö–Ω—è—è —ç–Ω–µ—Ä–≥–∏—è)
\end{itemize}
\end{examplebox}

\subsection{–°–æ–±—Å—Ç–≤–µ–Ω–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏}

\subsubsection{–°–≤—è–∑—ã–≤–∞—é—â–∞—è –ú–û}

–î–ª—è $E_1 = \frac{\alpha + \beta}{1 + S}$ –∏–∑ –ø–µ—Ä–≤–æ–≥–æ —É—Ä–∞–≤–Ω–µ–Ω–∏—è —Å–∏—Å—Ç–µ–º—ã:
\begin{equation}
c_1(\alpha - E_1) + c_2(\beta - E_1S) = 0
\end{equation}

–ü–æ—Å–∫–æ–ª—å–∫—É $\alpha - E_1 = \beta - E_1S$, –ø–æ–ª—É—á–∞–µ–º $c_1 = c_2$.

–ò–∑ —É—Å–ª–æ–≤–∏—è –Ω–æ—Ä–º–∏—Ä–æ–≤–∫–∏:
\begin{equation}
c_1^2 + c_2^2 + 2c_1c_2S = 1 \Rightarrow 2c_1^2(1 + S) = 1
\end{equation}

\begin{equation}
c_1 = c_2 = \frac{1}{\sqrt{2(1 + S)}}
\end{equation}

\textbf{–°–≤—è–∑—ã–≤–∞—é—â–∞—è –ú–û:}
\begin{equation}
\boxed{\psi_1 = \frac{1}{\sqrt{2(1 + S)}}(\phi_1 + \phi_2)}
\end{equation}

\subsubsection{–ê–Ω—Ç–∏—Å–≤—è–∑—ã–≤–∞—é—â–∞—è –ú–û}

–ê–Ω–∞–ª–æ–≥–∏—á–Ω–æ –¥–ª—è $E_2 = \frac{\alpha - \beta}{1 - S}$ –ø–æ–ª—É—á–∞–µ–º $c_1 = -c_2$:

\textbf{–ê–Ω—Ç–∏—Å–≤—è–∑—ã–≤–∞—é—â–∞—è –ú–û:}
\begin{equation}
\boxed{\psi_2 = \frac{1}{\sqrt{2(1 - S)}}(\phi_1 - \phi_2)}
\end{equation}

\section{–ö—Ä–∏—Ç–∏—á–µ—Å–∫–∏–π –∞–Ω–∞–ª–∏–∑ –ø—Ä–∏–±–ª–∏–∂–µ–Ω–∏—è}

\subsection{–ü–æ–¥—Å—Ç–∞–Ω–æ–≤–∫–∞ –≤ —É—Ä–∞–≤–Ω–µ–Ω–∏–µ –®—Ä—ë–¥–∏–Ω–≥–µ—Ä–∞}

\begin{warningbox}{–ö–ª—é—á–µ–≤–æ–π –≤–æ–ø—Ä–æ—Å}
–Ø–≤–ª—è–µ—Ç—Å—è –ª–∏ –ú–û-–õ–ö–ê–û —Ñ—É–Ω–∫—Ü–∏—è —Ç–æ—á–Ω—ã–º —Ä–µ—à–µ–Ω–∏–µ–º —É—Ä–∞–≤–Ω–µ–Ω–∏—è –®—Ä—ë–¥–∏–Ω–≥–µ—Ä–∞?

–ü—Ä–æ–≤–µ—Ä–∏–º: $\hat{H}\psi_{\text{–õ–ö–ê–û}} \stackrel{?}{=} E_{\text{–õ–ö–ê–û}}\psi_{\text{–õ–ö–ê–û}}$

\subsubsection{–ú–æ–ª–µ–∫—É–ª—è—Ä–Ω—ã–π –≥–∞–º–∏–ª—å—Ç–æ–Ω–∏–∞–Ω}

–î–ª—è H$_2^+$:
\begin{equation}
\hat{H} = -\frac{1}{2}\nabla^2 - \frac{1}{r_A} - \frac{1}{r_B} + \frac{1}{R_{AB}}
\end{equation}

\subsubsection{–î–µ–π—Å—Ç–≤–∏–µ –Ω–∞ –∞—Ç–æ–º–Ω—É—é –æ—Ä–±–∏—Ç–∞–ª—å}

\begin{align}
\hat{H}\phi_1 &= \left(-\frac{1}{2}\nabla^2 - \frac{1}{r_A} - \frac{1}{r_B} + \frac{1}{R_{AB}}\right)\phi_1 \\
&= \underbrace{-\frac{1}{2}\nabla^2\phi_1 - \frac{1}{r_A}\phi_1}_{E_{1s}\phi_1} + \underbrace{-\frac{1}{r_B}\phi_1 + \frac{1}{R_{AB}}\phi_1}_{\text{–≤–æ–∑–º—É—â–µ–Ω–∏–µ}}
\end{align}

–≥–¥–µ $E_{1s} = -\frac{1}{2}$ –∞.–µ. -- —ç–Ω–µ—Ä–≥–∏—è 1s –æ—Ä–±–∏—Ç–∞–ª–∏ –≤–æ–¥–æ—Ä–æ–¥–∞.

\subsubsection{–ü–æ–ª–Ω—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç}

\begin{align}
\hat{H}\psi_{\text{–õ–ö–ê–û}} &= \frac{1}{\sqrt{2(1+S)}}\bigl[E_{1s}(\phi_1 + \phi_2) \\
&\quad + \left(-\frac{1}{r_B} + \frac{1}{R_{AB}}\right)\phi_1 + \left(-\frac{1}{r_A} + \frac{1}{R_{AB}}\right)\phi_2\bigr]
\end{align}

\begin{importantbox}{–ö—Ä–∏—Ç–∏—á–µ—Å–∫–æ–µ –Ω–∞–±–ª—é–¥–µ–Ω–∏–µ}
